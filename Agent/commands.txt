# === Recode xtl_installer.py to UTF-8 (no BOM) and normalize CP1252 punctuation ===
$target = 'D:\Algo\Gold\wizard\xtl_installer.py'
if (!(Test-Path $target)) { throw "File not found: $target" }

# Backup
Copy-Item $target "$target.bak" -Force

# Encodings
$Utf8Strict = New-Object System.Text.UTF8Encoding($false, $true)  # no BOM, throw on invalid
$Utf8NoBom  = New-Object System.Text.UTF8Encoding($false)         # no BOM, no throw
$Cp1252     = [Text.Encoding]::GetEncoding(1252)

# Decode: strict UTF-8, else cp1252
$bytes = [IO.File]::ReadAllBytes($target)
try { $text = $Utf8Strict.GetString($bytes) } catch { $text = $Cp1252.GetString($bytes) }

# Normalize notorious cp1252 punctuation (— – ‘ ’ “ ” …) to ASCII
$text = $text `
  -replace "\u2014","-" `
  -replace "\u2013","-" `
  -replace "[\u2018\u2019]","'" `
  -replace "[\u201C\u201D]",'"' `
  -replace "\u2026","..."

# Ensure coding cookie on line 1 or 2 (after shebang if present)
$lines  = $text -split "(`r`n|`n|`r)"
$cookie = "# -*- coding: utf-8 -*-"
$hasTop = ($lines.Count -ge 1 -and $lines[0] -match "coding:\s*utf-8") -or
          ($lines.Count -ge 2 -and $lines[1] -match "coding:\s*utf-8")
if (-not $hasTop) {
  if ($lines[0].StartsWith("#!")) { $lines = @($lines[0], $cookie) + $lines[1..($lines.Count-1)] }
  else { $lines = @($cookie) + $lines }
}

# Normalize EOLs to CRLF and save as UTF-8 (no BOM)
$t = ($lines -join "`n") -replace "`r`n","`n" -replace "`r","`n"
$t = $t -replace "`n","`r`n"
[IO.File]::WriteAllText($target, $t, $Utf8NoBom)

# Verify strict UTF-8 now passes
[void]$Utf8Strict.GetString([IO.File]::ReadAllBytes($target))
Write-Host "OK: Rewrote as UTF-8 -> $target"
